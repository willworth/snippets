//Nested for loops
const myArray = [1,2,4,5,6,7,8,9,10]
const goalNum = 5;

const finder = function (arr){
  for (i=0; i<arr.length; i++){
    console.log('logging outer i =', i)
    for (j=0; j<arr.length;j++){
      if (i + j === goalNum){

      console.log(`${i} and ${j} are equal to ${goalNum}!`)
      }
    }
  }
}

finder(myArray)
